# в фиксированном массиве должен быть определен максимальный размер
# применяют, когда небольшой набор данных и его не нужно менять.
---
const MAX_ARRAY_SIZE = 2
var array [MAX_ARRAY_SIZE]string

array[0] = "foo"
array[1] = "bar"

fmt.Println(array[0])
fmt.Println(array[1])
fmt.Println(array)
===============================================
# автоматическое вычисление размера фиксированного массива
# Go может автоматически вычислить размер объявленного массива, указав тройную точку ... нотацию размера массива внутри квадратных скобок перед типом.
---
var flavours [4]string

fruits := [...]string{"Apple", "Mango", "Orange", "Banana"}
===============================================
# размер и тип фиксированного массива
# При назначении массива другой переменной массива в Go целевой массив должен быть того же типа и размера, что и исходный массив.
---
var flavours [6]string
fruits := [4]string{"apple", "mango", "kiwi", "avocado"}

flavours = fruits => error
==============================================
# при назначении одного фиксированного массива другому фиксированному массиву, он копирует все содержимое источника как новое значение. Любые изменения в старом массиве не будут зависеть от значений из нового массива.
